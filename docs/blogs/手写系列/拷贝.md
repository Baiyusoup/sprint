## 深拷贝
```javascript
function clone(obj) {
  if (Array.isArray(obj)) {
    // 处理数组类型
    return obj.map((item) => clone(item));
  } else if (obj && typeof obj === 'object') {
  // 处理对象类型
    const tmp = {};
    Object.keys(obj).forEach(key => {
      tmp[key] = clone(obj[key]);
    })
    return tmp;
  }
  
  // 基本数据类型，直接返回
  return obj;
}
```
